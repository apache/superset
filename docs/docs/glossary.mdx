import { Glossary } from '../src/resources/glossary.ts';
import KeyValue from '../src/components/KeyValue.tsx';
import { useColorMode } from '@docusaurus/theme-common';

export const GlossaryContent = () => {
  const { colorMode } = useColorMode();
  const isDark = colorMode === 'dark';
  
  return (
    <div>
      {Glossary.map(({ title, short, extended }) => (
        <KeyValue key={title} keyName={title} value={short} dark={isDark} />
      ))}
    </div>
  );
};
