import { glossary, GlossaryStructure } from '../../superset-frontend/packages/superset-ui-core/src/glossary';
import { Table, ConfigProvider } from 'antd';
import { theme } from "antd";
import { useColorMode } from '@docusaurus/theme-common';

export const GlossaryContent = () => {
  const { colorMode } = useColorMode();
  const isDark = colorMode === 'dark';
  
  return (
    <div>
      {Glossary.map(({ title, short, extended }) => (
        <KeyValue key={title} keyName={title} value={short} dark={isDark} />
                  <ConfigProvider
            theme={{
              algorithm: isDark
              ? theme.darkAlgorithm
              : theme.defaultAlgorithm,
            }}
          >
            <Table 
              dataSource={Glossary} 
              columns={GlossaryStructure}
              pagination={false}
              pagination={false}
              showHeader={false}
              bordered={true}
            />
          </ConfigProvider>
      ))}
    </div>
  );
};
