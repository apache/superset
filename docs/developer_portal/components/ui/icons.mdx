  const sizes = ['s', 'm', 'l', 'xl', 'xxl'];
  return (
    <div style={{ display: 'flex', gap: 24, alignItems: 'end' }}>
      {sizes.map(size => (
        <div key={size} style={{ textAlign: 'center' }}>

          <div style={{ fontSize: 12, marginTop: 8, color: '#666' }}>{size}</div>
        </div>
      ))}
    </div>
  );
}
