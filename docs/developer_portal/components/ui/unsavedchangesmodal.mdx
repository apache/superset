  return (
    <div>

      </Button>
      <UnsavedChangesModal
        showModal={show}
        onHide={() => setShow(false)}
        handleSave={() => { alert('Saved!'); setShow(false); }}
        onConfirmNavigation={() => { alert('Discarded changes'); setShow(false); }}
        title="Unsaved Changes"

      </UnsavedChangesModal>
    </div>
  );
}
